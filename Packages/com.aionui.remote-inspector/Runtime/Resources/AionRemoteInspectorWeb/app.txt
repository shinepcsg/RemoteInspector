async function boot() {
  bindEvents();
  setupTabs();

  const origin = location.origin && location.origin !== "null" ? location.origin : "http://127.0.0.1:7759";
  elements.hostInput.value = origin;
  elements.selectionTitle.textContent = "Inspector";
  elements.selectionMeta.textContent = "No selection";

  try {
    const response = await fetch("/api/info");
    if (response.ok) {
      state.info = await response.json();
      elements.brandSubtitle.textContent = `${state.info.displayName} · ${state.info.localUrl}`;
    }
  } catch {
    // The UI still works when /api/info is not available yet.
  }
}

function bindEvents() {
  elements.connectButton.addEventListener("click", connect);
  elements.refreshButton.addEventListener("click", refreshAll);
  elements.hierarchySearch.addEventListener("input", debounce(refreshHierarchy, 180));
  elements.inactiveToggle.addEventListener("change", refreshHierarchy);
  elements.hiddenToggle.addEventListener("change", refreshHierarchy);
  elements.createEmptyButton.addEventListener("click", createEmpty);
  elements.duplicateButton.addEventListener("click", duplicateSelection);
  elements.deleteButton.addEventListener("click", deleteSelection);
  elements.componentAddButton.addEventListener("click", addComponentToSelection);
  elements.clearLogsButton.addEventListener("click", clearLogs);
  elements.logSearch.addEventListener("input", renderLogs);
  elements.consoleSendButton.addEventListener("click", runConsoleCommand);
  elements.consoleInput.addEventListener("keydown", (event) => {
    if (event.key === "Enter") {
      event.preventDefault();
      runConsoleCommand();
    }
  });
}

function setupTabs() {
  const buttons = Array.from(document.querySelectorAll(".tab-button"));
  const panels = Array.from(document.querySelectorAll(".tab-panel"));
  for (const button of buttons) {
    button.addEventListener("click", () => {
      const tab = button.dataset.tab;
      for (const other of buttons) {
        other.classList.toggle("is-active", other === button);
      }
      for (const panel of panels) {
        panel.classList.toggle("is-active", panel.id === `tab-${tab}`);
      }
    });
  }
}

async function connect() {
  disconnect();
  const host = normalizeHttpHost(elements.hostInput.value);
  const socketUrl = host.replace(/^http/i, "ws") + "/ws";
  setStatus("Connecting", "#c89454");

  try {
    const socket = new WebSocket(socketUrl);
    socket.addEventListener("open", () => {
      state.socket = socket;
      send("auth", { password: elements.passwordInput.value || "" })
        .then((payload) => {
          if (!payload.success) {
            showToast(payload.message || "Authentication failed.", true);
            disconnect();
            return;
          }

          state.authenticated = true;
          state.info = payload.info || state.info;
          elements.brandSubtitle.textContent = `${state.info.displayName} · ${state.info.lanUrl || state.info.localUrl}`;
          setStatus("Connected", "#6fa06f");
          showToast(payload.message || "Connected.");
          refreshAll();
        })
        .catch((error) => {
          showToast(error.message, true);
          disconnect();
        });
    });

    socket.addEventListener("message", onSocketMessage);
    socket.addEventListener("close", () => {
      setStatus("Disconnected", "#bb6666");
      state.authenticated = false;
    });
    socket.addEventListener("error", () => {
      showToast("WebSocket connection failed.", true);
      setStatus("Error", "#bb6666");
    });
  } catch (error) {
    showToast(error.message, true);
    setStatus("Error", "#bb6666");
  }
}

function disconnect() {
  if (state.socket) {
    try {
      state.socket.close();
    } catch {
      // Ignore close errors.
    }
  }

  state.socket = null;
  state.authenticated = false;
  for (const pending of state.pending.values()) {
    pending.reject(new Error("Socket closed."));
  }
  state.pending.clear();
}

function onSocketMessage(event) {
  let message;
  try {
    message = JSON.parse(event.data);
  } catch {
    return;
  }

  let payload = {};
  try {
    payload = message.payloadJson ? JSON.parse(message.payloadJson) : {};
  } catch {
    payload = {};
  }

  let handledByPending = false;
  if (message.requestId && state.pending.has(message.requestId)) {
    const pending = state.pending.get(message.requestId);
    state.pending.delete(message.requestId);
    handledByPending = true;
    if (message.type === "error") {
      pending.reject(new Error(payload.message || "Request failed."));
    } else {
      pending.resolve(payload);
    }
  }

  switch (message.type) {
    case "info/result":
      state.info = payload;
      break;
    case "hierarchy/result":
      state.hierarchy = payload.nodes || [];
      syncHierarchyCollapseState();
      if (state.selectedGameObjectId && !state.hierarchy.some((node) => node.instanceId === state.selectedGameObjectId)) {
        state.selectedGameObjectId = 0;
        state.inspector = null;
        renderInspector();
      }
      renderHierarchy();
      break;
    case "inspect/result":
      state.inspector = payload;
      renderInspector();
      break;
    case "logs/result":
      state.logs = payload.entries || [];
      renderLogs();
      break;
    case "log/append":
      state.logs.push(payload);
      renderLogs();
      break;
    case "logs/cleared":
      state.logs = [];
      state.selectedLogIndex = 0;
      renderLogs();
      break;
    case "scene/changed":
      refreshHierarchy();
      if (state.selectedGameObjectId) {
        refreshInspector();
      }
      break;
    case "ack":
      showToast(payload.message || "Updated.");
      if (state.selectedGameObjectId) {
        refreshInspector();
      }
      break;
    case "error":
      if (!handledByPending) {
        showToast(payload.message || "Request failed.", true);
      }
      break;
    default:
      break;
  }
}

function send(type, payload) {
  if (!state.socket || state.socket.readyState !== WebSocket.OPEN) {
    return Promise.reject(new Error("WebSocket is not connected."));
  }

  const requestId = String(state.requestId++);
  const envelope = {
    type,
    requestId,
    payloadJson: payload ? JSON.stringify(payload) : ""
  };

  state.socket.send(JSON.stringify(envelope));
  return new Promise((resolve, reject) => {
    const timeout = window.setTimeout(() => {
      state.pending.delete(requestId);
      reject(new Error(`Timed out waiting for ${type}.`));
    }, 6000);

    state.pending.set(requestId, {
      resolve: (result) => {
        window.clearTimeout(timeout);
        resolve(result);
      },
      reject
    });
  });
}

function refreshAll() {
  if (!state.authenticated) {
    return;
  }

  refreshHierarchy();
  refreshLogs();
  if (state.selectedGameObjectId) {
    refreshInspector();
  } else {
    renderInspector();
  }
}

function refreshHierarchy() {
  if (!state.authenticated) {
    return;
  }

  send("hierarchy/get", {
    query: elements.hierarchySearch.value || "",
    includeInactive: elements.inactiveToggle.checked,
    showHidden: elements.hiddenToggle.checked
  }).catch((error) => showToast(error.message, true));
}

function refreshInspector() {
  if (!state.authenticated || !state.selectedGameObjectId) {
    return;
  }

  send("inspect/get", {
    gameObjectInstanceId: state.selectedGameObjectId
  }).catch((error) => showToast(error.message, true));
}

function refreshLogs() {
  if (!state.authenticated) {
    return;
  }

  send("logs/get", {}).catch((error) => showToast(error.message, true));
}

function renderHierarchy() {
  const visibleNodes = getVisibleHierarchyNodes();
  elements.hierarchySummary.textContent = visibleNodes.length === state.hierarchy.length
    ? `${state.hierarchy.length} object(s)`
    : `${visibleNodes.length} / ${state.hierarchy.length} object(s)`;
  elements.hierarchyList.innerHTML = "";

  if (!state.hierarchy.length) {
    const empty = document.createElement("div");
    empty.className = "empty-state";
    empty.textContent = "No objects found.";
    elements.hierarchyList.appendChild(empty);
    return;
  }

  const fragment = document.createDocumentFragment();
  for (const entry of visibleNodes) {
    const { node, hasChildren, collapsed } = entry;

    const button = document.createElement("button");
    button.type = "button";
    button.className = "hierarchy-row";
    if (node.instanceId === state.selectedGameObjectId) {
      button.classList.add("is-selected");
    }
    if (!node.activeInHierarchy) {
      button.classList.add("is-inactive");
    }

    button.style.paddingLeft = `${6 + node.depth * 14}px`;
    button.title = `${node.sceneName} · [${node.instanceId}]${node.hidden ? " · Hidden" : ""}`;
    button.innerHTML = `
      <span class="hierarchy-arrow hierarchy-arrow-toggle ${hasChildren ? (collapsed ? "is-collapsed" : "is-open") : "is-empty"}">${hasChildren ? (collapsed ? "▸" : "▾") : "·"}</span>
      <span class="hierarchy-icon"></span>
      <span class="hierarchy-item-main">
        <span class="hierarchy-name">${escapeHtml(node.name)}</span>
      </span>
    `;
    button.addEventListener("click", (event) => {
      if (hasChildren && event.target instanceof Element && event.target.closest(".hierarchy-arrow-toggle")) {
        toggleHierarchyNode(node.instanceId);
        return;
      }

      state.selectedGameObjectId = node.instanceId;
      refreshInspector();
      renderHierarchy();
    });
    fragment.appendChild(button);
  }

  elements.hierarchyList.appendChild(fragment);
}

function getVisibleHierarchyNodes() {
  const visibleNodes = [];
  const collapsedDepths = [];
  for (let index = 0; index < state.hierarchy.length; index += 1) {
    const node = state.hierarchy[index];
    while (collapsedDepths.length && node.depth <= collapsedDepths[collapsedDepths.length - 1]) {
      collapsedDepths.pop();
    }

    if (collapsedDepths.length) {
      continue;
    }

    const hasChildren = hasHierarchyChildren(index);
    const collapsed = hasChildren && state.collapsedHierarchyNodes.has(node.instanceId);
    visibleNodes.push({ node, index, hasChildren, collapsed });
    if (collapsed) {
      collapsedDepths.push(node.depth);
    }
  }

  return visibleNodes;
}

function hasHierarchyChildren(index) {
  const node = state.hierarchy[index];
  const nextNode = state.hierarchy[index + 1];
  return Boolean(node && nextNode && nextNode.depth > node.depth);
}

function toggleHierarchyNode(instanceId) {
  if (state.collapsedHierarchyNodes.has(instanceId)) {
    state.collapsedHierarchyNodes.delete(instanceId);
  } else {
    state.collapsedHierarchyNodes.add(instanceId);
  }

  renderHierarchy();
}

function syncHierarchyCollapseState() {
  const validIds = new Set();
  for (let index = 0; index < state.hierarchy.length; index += 1) {
    if (hasHierarchyChildren(index)) {
      validIds.add(state.hierarchy[index].instanceId);
    }
  }

  for (const instanceId of Array.from(state.collapsedHierarchyNodes)) {
    if (!validIds.has(instanceId)) {
      state.collapsedHierarchyNodes.delete(instanceId);
    }
  }
}

function renderInspector() {
  elements.selectionTitle.textContent = "Inspector";
  if (!state.inspector) {
    elements.selectionMeta.textContent = state.selectedGameObjectId ? `[${state.selectedGameObjectId}]` : "No selection";
    elements.inspectorView.textContent = "Select a GameObject from the hierarchy.";
    elements.inspectorView.classList.add("empty-state");
    return;
  }

  elements.selectionMeta.textContent = `${state.inspector.sceneName || "(No Scene)"} · [${state.inspector.gameObjectInstanceId}]`;
  elements.inspectorView.innerHTML = "";
  elements.inspectorView.classList.remove("empty-state");

  const stack = document.createElement("div");
  stack.className = "inspector-stack";
  stack.appendChild(buildObjectHeaderCard(state.inspector));

  for (const component of state.inspector.components || []) {
    stack.appendChild(buildComponentSection(component));
  }

  stack.appendChild(buildAddComponentRow());
  elements.inspectorView.appendChild(stack);
}

function buildComponentSection(component) {
  const sectionKey = `component:${component.instanceId}`;
  const collapsed = state.collapsedSections.has(sectionKey);

  const card = document.createElement("section");
  card.className = "component-section";
  if (collapsed) {
    card.classList.add("is-collapsed");
  }

  const header = document.createElement("div");
  header.className = "component-section-header";

  const toggle = document.createElement("button");
  toggle.type = "button";
  toggle.className = "section-toggle";
  toggle.innerHTML = `
    <span class="foldout-arrow">${collapsed ? "▸" : "▾"}</span>
    <span class="component-section-title">${escapeHtml(component.typeName)}</span>
  `;
  toggle.addEventListener("click", () => toggleSection(sectionKey));
  header.appendChild(toggle);

  const meta = document.createElement("span");
  meta.className = "component-section-meta";
  meta.textContent = `[${component.instanceId}]`;
  header.appendChild(meta);

  const actions = document.createElement("div");
  actions.className = "section-actions";
  if (component.canToggleEnabled && !component.enabled) {
    const chip = document.createElement("span");
    chip.className = "section-chip";
    chip.textContent = "Disabled";
    actions.appendChild(chip);
  }
  if (component.canDestroy) {
    const removeButton = document.createElement("button");
    removeButton.type = "button";
    removeButton.className = "danger";
    removeButton.textContent = "Remove";
    removeButton.addEventListener("click", () => {
      send("component/destroy", { componentInstanceId: component.instanceId })
        .catch((error) => showToast(error.message, true));
    });
    actions.appendChild(removeButton);
  }
  header.appendChild(actions);
  card.appendChild(header);

  const body = document.createElement("div");
  body.className = "component-body";
  const members = component.members || [];
  const hiddenPaths = new Set();
  for (const member of members) {
    if (hiddenPaths.has(member.path)) {
      continue;
    }

    body.appendChild(buildPropertyRow(member, members, hiddenPaths, component.instanceId));
  }
  card.appendChild(body);

  return card;
}

function buildPropertyRow(member, allMembers, hiddenPaths, componentInstanceId) {
  const row = document.createElement("div");
  row.className = `property-row property-indent-${Math.min(member.depth || 0, 5)}`;

  const label = document.createElement("div");
  label.className = "property-label";
  label.innerHTML = `
    <div class="property-label-inner">
      <span class="property-label-text">${escapeHtml(member.label)}</span>
    </div>
  `;
  row.appendChild(label);

  const controlCell = document.createElement("div");
  controlCell.className = "property-control";
  const inline = document.createElement("div");
  inline.className = "property-inline";

  if (member.editable) {
    inline.appendChild(buildEditableControl(member, allMembers, hiddenPaths, componentInstanceId));
  } else {
    inline.appendChild(buildReadonlyControl(member));
  }

  const typeTag = document.createElement("span");
  typeTag.className = "property-type-tag";
  typeTag.textContent = member.typeName;
  inline.appendChild(typeTag);

  controlCell.appendChild(inline);
  row.appendChild(controlCell);
  return row;
}

function buildEditableControl(member, allMembers, hiddenPaths, componentInstanceId) {
  const composite = tryBuildCompositeControl(member, allMembers, hiddenPaths, componentInstanceId);
  if (composite) {
    return composite;
  }

  if (isBooleanMember(member)) {
    const wrapper = document.createElement("label");
    wrapper.className = "property-checkbox-line";

    const input = document.createElement("input");
    input.type = "checkbox";
    input.className = "field-checkbox";
    input.checked = member.value === "true";
    input.dataset.lastCommitted = input.checked ? "true" : "false";
    input.addEventListener("change", () => {
      commitMemberValue(componentInstanceId, member, input.checked ? "true" : "false", input);
    });
    wrapper.appendChild(input);
    return wrapper;
  }

  if (hasEnumOptions(member)) {
    const select = document.createElement("select");
    select.className = "property-select";
    for (const optionValue of member.options || []) {
      const option = document.createElement("option");
      option.value = optionValue;
      option.textContent = optionValue;
      select.appendChild(option);
    }
    select.value = member.value;
    bindInputCommit(select, member, componentInstanceId);
    return select;
  }

  if (member.multiline) {
    const textarea = document.createElement("textarea");
    textarea.className = "property-textarea";
    textarea.value = member.value === "null" ? "" : member.value;
    bindInputCommit(textarea, member, componentInstanceId);
    return textarea;
  }

  const input = document.createElement("input");
  input.type = isNumericMember(member) ? "number" : "text";
  input.className = "property-input";
  if (input.type === "number") {
    input.step = isIntegerMember(member) ? "1" : "any";
  }
  input.value = member.value === "null" ? "" : member.value;
  bindInputCommit(input, member, componentInstanceId);
  return input;
}

function buildReadonlyControl(member) {
  if ((member.controlHint || "") === "object") {
    const objectRef = document.createElement("div");
    objectRef.className = "object-reference-field";
    objectRef.textContent = member.value === "null"
      ? `None (${member.referenceTypeName || member.typeName || "Object"})`
      : member.value;
    return objectRef;
  }

  const readonly = document.createElement("div");
  readonly.className = "property-value-readonly";
  readonly.textContent = member.value;
  return readonly;
}

function tryBuildCompositeControl(member, allMembers, hiddenPaths, componentInstanceId) {
  const descriptor = getCompositeDescriptor(member);
  if (!descriptor) {
    return null;
  }

  const childMembers = descriptor.suffixes
    .map((suffix) => findMember(allMembers, `${member.path}${suffix}`))
    .filter(Boolean);

  if (childMembers.length !== descriptor.suffixes.length) {
    return null;
  }

  for (const child of childMembers) {
    hiddenPaths.add(child.path);
  }

  return buildCompositeControl(member, childMembers, descriptor, componentInstanceId);
}

function buildCompositeControl(member, childMembers, descriptor, componentInstanceId) {
  const wrapper = document.createElement("div");
  wrapper.className = `composite-field composite-${descriptor.kind}`;

  if (descriptor.kind === "color") {
    const preview = document.createElement("div");
    preview.className = "color-preview";
    wrapper.appendChild(preview);
  }

  const inputs = [];
  for (let index = 0; index < childMembers.length; index += 1) {
    const child = childMembers[index];
    const cell = document.createElement("label");
    cell.className = "composite-cell";

    const axis = document.createElement("span");
    axis.className = `composite-axis axis-${descriptor.axisLabels[index].toLowerCase()}`;
    axis.textContent = descriptor.axisLabels[index];
    cell.appendChild(axis);

    const input = document.createElement("input");
    input.type = "number";
    input.className = "composite-input";
    input.step = descriptor.integer ? "1" : "any";
    input.value = child.value === "null" ? "" : child.value;
    cell.appendChild(input);

    wrapper.appendChild(cell);
    inputs.push(input);
  }

  const serialize = () => `[${inputs.map((input) => normalizeCompositePart(input.value)).join(", ")}]`;
  wrapper.dataset.lastCommitted = serialize();

  const commit = () => {
    const nextValue = serialize();
    if ((wrapper.dataset.lastCommitted || "") === nextValue) {
      return;
    }

    wrapper.dataset.lastCommitted = nextValue;
    updateMember(componentInstanceId, member.path, nextValue);
  };

  for (const input of inputs) {
    input.addEventListener("change", commit);
    input.addEventListener("blur", commit);
    input.addEventListener("input", () => updateCompositePreview(wrapper, descriptor, inputs));
    input.addEventListener("keydown", (event) => {
      if (event.key === "Enter") {
        event.preventDefault();
        commit();
      }
    });
  }

  updateCompositePreview(wrapper, descriptor, inputs);
  return wrapper;
}

function updateCompositePreview(wrapper, descriptor, inputs) {
  if (descriptor.kind !== "color") {
    return;
  }

  const preview = wrapper.querySelector(".color-preview");
  if (!preview) {
    return;
  }

  const values = inputs.map((input) => Number.parseFloat(input.value || "0"));
  const divisor = descriptor.integer ? 255 : 1;
  const r = clamp01((values[0] || 0) / divisor);
  const g = clamp01((values[1] || 0) / divisor);
  const b = clamp01((values[2] || 0) / divisor);
  const a = clamp01(((values[3] ?? (descriptor.integer ? 255 : 1)) || 0) / divisor);
  preview.style.background = `rgba(${Math.round(r * 255)}, ${Math.round(g * 255)}, ${Math.round(b * 255)}, ${a})`;
}

function getCompositeDescriptor(member) {
  switch (member.controlHint) {
    case "vector2":
      return { kind: "vector", integer: false, suffixes: [".x", ".y"], axisLabels: ["X", "Y"] };
    case "vector2int":
      return { kind: "vector", integer: true, suffixes: [".x", ".y"], axisLabels: ["X", "Y"] };
    case "vector3":
      return { kind: "vector", integer: false, suffixes: [".x", ".y", ".z"], axisLabels: ["X", "Y", "Z"] };
    case "vector3int":
      return { kind: "vector", integer: true, suffixes: [".x", ".y", ".z"], axisLabels: ["X", "Y", "Z"] };
    case "vector4":
      return { kind: "vector", integer: false, suffixes: [".x", ".y", ".z", ".w"], axisLabels: ["X", "Y", "Z", "W"] };
    case "quaternion":
      return { kind: "vector", integer: false, suffixes: [".eulerAngles.x", ".eulerAngles.y", ".eulerAngles.z"], axisLabels: ["X", "Y", "Z"] };
    case "color":
      return { kind: "color", integer: member.typeName === "Color32", suffixes: [".r", ".g", ".b", ".a"], axisLabels: ["R", "G", "B", "A"] };
    case "rect":
      return { kind: "rect", integer: false, suffixes: [".x", ".y", ".width", ".height"], axisLabels: ["X", "Y", "W", "H"] };
    case "rectint":
      return { kind: "rect", integer: true, suffixes: [".x", ".y", ".width", ".height"], axisLabels: ["X", "Y", "W", "H"] };
    default:
      return null;
  }
}

function normalizeCompositePart(value) {
  const trimmed = (value || "").trim();
  return trimmed === "" ? "0" : trimmed;
}

function bindInputCommit(input, member, componentInstanceId) {
  input.dataset.lastCommitted = member.value === "null" ? "" : member.value;

  const commit = () => {
    commitMemberValue(componentInstanceId, member, input.value, input);
  };

  input.addEventListener("change", commit);
  input.addEventListener("blur", commit);
  input.addEventListener("keydown", (event) => {
    if (input.tagName === "TEXTAREA") {
      if ((event.ctrlKey || event.metaKey) && event.key === "Enter") {
        event.preventDefault();
        commit();
      }
      return;
    }

    if (event.key === "Enter") {
      event.preventDefault();
      commit();
    }
  });
}

function commitMemberValue(componentInstanceId, member, nextValue, inputElement) {
  const normalizedValue = nextValue ?? "";
  if ((inputElement.dataset.lastCommitted || "") === normalizedValue) {
    return;
  }

  inputElement.dataset.lastCommitted = normalizedValue;
  updateMember(componentInstanceId, member.path, normalizedValue);
}

function updateMember(componentInstanceId, memberPath, value) {
  if (!state.selectedGameObjectId) {
    return;
  }

  send("member/set", {
    gameObjectInstanceId: state.selectedGameObjectId,
    componentInstanceId: componentInstanceId || 0,
    memberPath,
    value
  }).catch((error) => showToast(error.message, true));
}

  const componentType = elements.componentAddInput.value.trim();
  if (!componentType) {

    gameObjectInstanceId: state.selectedGameObjectId,

function renderLogs() {
  const query = (elements.logSearch.value || "").toLowerCase();
  const filtered = state.logs.filter((entry) => {
    if (!query) {
      return true;
    }

    return `${entry.logType} ${entry.message} ${entry.stackTrace}`.toLowerCase().includes(query);
  });

  elements.logsSummary.textContent = `${filtered.length} / ${state.logs.length} entries`;
  elements.logList.innerHTML = "";

  const selectedEntry = filtered.find((entry) => entry.index === state.selectedLogIndex);
  if (!selectedEntry && filtered.length) {
    state.selectedLogIndex = filtered[0].index;
  }

  for (const entry of filtered) {
    const item = document.createElement("div");
    const type = (entry.logType || "Log").toLowerCase();
    item.className = `log-item ${type}`;
    if (entry.index === state.selectedLogIndex) {
      item.classList.add("is-selected");
    }

    item.innerHTML = `
      <div class="log-meta">${escapeHtml(entry.timestamp || "")} · ${escapeHtml(entry.logType || "")}</div>
      <div>${escapeHtml(entry.message || "")}</div>
    `;
    item.addEventListener("click", () => {
      state.selectedLogIndex = entry.index;
      renderLogs();
    });
    elements.logList.appendChild(item);
  }

  const detailEntry = filtered.find((entry) => entry.index === state.selectedLogIndex);
  if (!detailEntry) {
    elements.logDetail.textContent = filtered.length ? "Select a log entry." : "No matching log entries.";
    return;
  }

  elements.logDetail.textContent = `${detailEntry.message || ""}\n\n${detailEntry.stackTrace || ""}`.trim();
}

function runConsoleCommand() {
  const command = elements.consoleInput.value.trim();
  if (!command) {
    return;
  }

  send("console/execute", { command })
    .then((payload) => {
      appendConsole(payload.success ? "info" : "error", command, payload.output || "");
      elements.consoleInput.value = "";
    })
    .catch((error) => showToast(error.message, true));
}

function appendConsole(kind, command, output) {
  const line = document.createElement("div");
  line.className = `console-line ${kind}`;
  line.innerHTML = `<div class="prompt">&gt; ${escapeHtml(command)}</div><div>${escapeHtml(output || "")}</div>`;
  elements.consoleOutput.prepend(line);
}

function toggleSection(sectionKey) {
  if (state.collapsedSections.has(sectionKey)) {
    state.collapsedSections.delete(sectionKey);
  } else {
    state.collapsedSections.add(sectionKey);
  }
  renderInspector();
}

function findMember(members, path) {
  return (members || []).find((member) => member.path === path) || null;
}

function isBooleanMember(member) {
  return member && (member.typeName === "Boolean" || member.typeName === "bool");
}

function hasEnumOptions(member) {
  return Boolean(member && Array.isArray(member.options) && member.options.length);
}

function isNumericMember(member) {
  return Boolean(member && ["number", "integer", "layermask"].includes(member.controlHint));
}

function isIntegerMember(member) {
  return Boolean(member && ["integer", "layermask"].includes(member.controlHint));
}

function clamp01(value) {
  return Math.min(1, Math.max(0, Number.isFinite(value) ? value : 0));
}

function normalizeHttpHost(value) {
  const trimmed = (value || "").trim();
  if (!trimmed) {
    return "http://127.0.0.1:7759";
  }

  return trimmed.endsWith("/") ? trimmed.slice(0, -1) : trimmed;
}

let toastTimer = 0;
function showToast(message, isError = false) {
  elements.toast.textContent = message;
  elements.toast.classList.toggle("is-error", isError);
  elements.toast.classList.add("is-visible");
  window.clearTimeout(toastTimer);
  toastTimer = window.setTimeout(() => elements.toast.classList.remove("is-visible"), 2200);
}

function debounce(callback, delay) {
  let timer = 0;
  return (...args) => {
    window.clearTimeout(timer);
    timer = window.setTimeout(() => callback(...args), delay);
  };
}

function escapeHtml(value) {
  return String(value || "")
    .replaceAll("&", "&amp;")
    .replaceAll("<", "&lt;")
    .replaceAll(">", "&gt;");
}
